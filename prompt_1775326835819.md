import React from 'react';

interface TastProps {
  message?: string;
}

const Tast: React.FC<TastProps> = ({ message = "Hello from Tast Component!" }) => {
  return (
    <div style={{ padding: '20px', border: '1px solid #ccc', borderRadius: '5px' }}>
      <h2>Tast Component</h2>
      <p>{message}</p>
      <button onClick={() => alert(`Button clicked! Message: ${message}`)}>
        Click Me
      </button>
    </div>
  );
};

export default Tast;
